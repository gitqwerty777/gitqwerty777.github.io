簡介

採用簡潔的模板語法來宣告式地將資料渲染進 DOM 的系統

<!-- html part -->
<div id="app">
  {{ message }}
</div>
//js part
var app = new Vue({
  el: '#app',
  data: {
    message: 'Hello Vue.js!'
    todo: []
  },
  methods: {
    reverseMessage: function () {
      this.message = this.message.split('').reverse().join('')
    }
  }
})
